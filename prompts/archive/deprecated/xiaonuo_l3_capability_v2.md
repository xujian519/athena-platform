#### 工具选择算法

```python
def select_tools(intent, context, user_id):
    """
    智能工具选择算法
    """
    # 步骤1: 候选工具过滤
    candidates = filter_by_intent(all_tools, intent)

    # 步骤2: 评分计算
    for tool in candidates:
        # 性能评分
        performance_score = tool.performance_score

        # 上下文相关性
        context_score = calculate_context_relevance(tool, intent, context)

        # 用户偏好
        preference_score = get_user_preference(user_id, tool.name)

        # 工具适用性
        suitability_score = calculate_suitability(tool, context)

        # 综合评分
        final_score = (
            performance_score * 0.30 +
            context_score * 0.30 +
            preference_score * 0.20 +
            suitability_score * 0.20
        )

        tool.final_score = final_score

    # 步骤3: 排序并返回top-k
    return sorted(candidates, key=lambda x: x.final_score, reverse=True)[:3]
```
