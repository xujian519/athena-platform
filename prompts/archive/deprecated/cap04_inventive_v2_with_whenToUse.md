### ⚡ Turn 1: 并行检索（所有独立操作同时进行）

```python
# 第一步：确定最接近的现有技术（并行检索）
async def find_closest_prior_art_parallel(target_patent):
    """并行查找最接近的现有技术"""

    # 并行执行所有独立检索
    results = await parallel([
        # 检索1: PostgreSQL精确检索（相同IPC）
        query_postgresql("""
            SELECT patent_number, application_date, title, abstract, ipc_main_class
            FROM patents
            WHERE ipc_main_class = %s
              AND application_date < %s
            ORDER BY application_date DESC
            LIMIT 50
        """, (target_patent["ipc_main_class"], target_patent["application_date"])),

        # 检索2: Qdrant向量检索（语义相似）
        qdrant_search(
            collection_name="patent_full_text",
            query_vector=embed(target_patent["title"] + " " + target_patent["abstract"]),
            top_k=20
        ),

        # 检索3: Neo4j图谱查询（技术关联）
        query_neo4j("""
            MATCH (p:Patent)-[:IN_IPC_CLASS]->(ipc:IPC)
            WHERE ipc.code = $ipc_code
            MATCH (p)-[:CITES]->(cited:Patent)
            RETURN cited.patent_number, cited.title, cited.abstract
            LIMIT 30
        """, {"ipc_code": target_patent["ipc_main_class"]})
    ])

    # 合并结果
    postgres_results, qdrant_results, neo4j_results = results
    candidates = merge_and_rank(postgres_results, qdrant_results, neo4j_results)

    return candidates
```
