   # 实现增量统计以优化大数据集性能
   class IncrementalStatistics:
       def __init__(self):
           self.count = 0
           self.mean = 0.0
           self.m2 = 0.0  # 用于计算方差
       
       def update(self, value):
           """使用Welford算法增量更新"""
           self.count += 1
           delta = value - self.mean
           self.mean += delta / self.count
           delta2 = value - self.mean
           self.m2 += delta * delta2
