```python
class QualityDimension(Enum):
    TECHNICAL_VALUE = "technical_value"       # 技术价值 (20%)
    LEGAL_STABILITY = "legal_stability"       # 法律稳定性 (20%)
    COMMERCIAL_VALUE = "commercial_value"     # 商业价值 (15%)
    SCOPE_CLARITY = "scope_clarity"           # 权利要求清晰度 (15%)
    DISCLOSURE_QUALITY = "disclosure_quality" # 说明书质量 (10%)
    INNOVATION_LEVEL = "innovation_level"     # 创新程度 (10%)
    ENFORCEABILITY = "enforceability"         # 可执行性 (5%)
    MARKET_RELEVANCE = "market_relevance"     # 市场相关性 (5%)
```

### 3. 质量等级定义

```python
class QualityGrade(Enum):
    EXCELLENT = "A+"      # 优秀 (>90)
    VERY_GOOD = "A"       # 很好 (80-90)
    GOOD = "B+"           # 好 (70-80)
    AVERAGE = "B"         # 一般 (60-70)
    BELOW_AVERAGE = "C+"  # 中下 (50-60)
    POOR = "C"            # 较差 (40-50)
    VERY_POOR = "D"       # 很差 (30-40)
    CRITICAL = "F"        # 严重问题 (<30)
```

### 4. 评估类型

```python
class AssessmentType(Enum):
    FULL = "full"             # 完整评估 (8个维度)
    QUICK = "quick"           # 快速评估 (3个核心维度)
    CLAIMS_ONLY = "claims"    # 仅权利要求
    TECHNICAL = "technical"   # 技术评估
    LEGAL = "legal"           # 法律评估
    COMMERCIAL = "commercial" # 商业评估
```

### 5. 核心数据结构

```python
# 维度分数
@dataclass
class DimensionScore:
    dimension: QualityDimension
    score: float                    # 0-100
    weight: float                   # 权重
    weighted_score: float           # 加权分数
    confidence: float               # 置信度
    analysis: str                   # 分析说明
    key_factors: List[str]          # 关键因素

# 质量风险
@dataclass
class QualityRisk:
    risk_id: str
    risk_type: str
    dimension: QualityDimension
    severity: RiskLevel
    description: str
    impact: str
    mitigation: str
    likelihood: float

# 改进建议
@dataclass
class ImprovementSuggestion:
    suggestion_id: str
    dimension: QualityDimension
    priority: ImprovementPriority
    title: str
    description: str
    action_items: List[str]
    expected_improvement: float
    effort_level: str
    timeline: str

# 评估结果
@dataclass
class EnhancedQualityAssessment:
    assessment_id: str
    patent_number: str
    assessment_type: AssessmentType
    overall_score: float
    overall_grade: QualityGrade
    dimension_scores: List[DimensionScore]
    risks: List[QualityRisk]
    improvements: List[ImprovementSuggestion]
    benchmarks: List[BenchmarkComparison]
    predicted_validity: float
    predicted_enforceability: float
    predicted_litigation_risk: float
```

## 使用示例

### 完整评估

```python
from core.patent.ai_services.quality_assessment_enhanced import (
    EnhancedQualityAssessor,
    AssessmentType,
    format_assessment_report
)

# 创建评估器
assessor = EnhancedQualityAssessor()

# 准备专利数据
patent_data = {
    "technical_features": ["特征1", "特征2", "特征3"],
    "technical_field": "人工智能",
    "embodiments": ["实施例1", "实施例2"],
    "claims": [
        {"type": "independent", "text": "一种方法..."},
        {"type": "dependent", "text": "根据权利要求1..."}
    ],
    "description": "说明书内容" * 100,
    "figures": ["图1", "图2", "图3"],
    "keywords": ["新的", "改进"]
}

# 执行评估
result = await assessor.assess(
    patent_number="CN1234567A",
    patent_data=patent_data,
    assessment_type=AssessmentType.FULL
)

# 输出报告
print(format_assessment_report(result))
```

### 快速评估

```python
from core.patent.ai_services.quality_assessment_enhanced import (
    assess_patent_quality,
    AssessmentType
)

result = await assess_patent_quality(
    patent_number="CN1234567A",
    patent_data=patent_data,
    assessment_type=AssessmentType.QUICK
)

print(f"得分: {result.overall_score:.1f}")
print(f"等级: {result.overall_grade.value}")
```
