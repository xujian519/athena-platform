    TEXT_ONLY = "text_only"           # 纯文本检索

class FusionStrategy(Enum):

## 使用示例

### 基本检索

```python
from core.patent.ai_services.multimodal_retrieval import (
    MultimodalRetrievalSystem,
    RetrievalConfig,
    SearchMode,
    format_search_result
)

# 创建系统
config = RetrievalConfig(mode=SearchMode.HYBRID, top_k=20)
system = MultimodalRetrievalSystem(config=config)

# 执行检索
result = await system.search("光伏充电控制器")

# 查看结果
print(format_search_result(result))
```

### 多模态检索

```python
from core.patent.ai_services.multimodal_retrieval import (
    multimodal_search
)

# 使用图像和文本检索
query_image = open("product.png", "rb").read()
result = await multimodal_search(
    query="电路结构",
    query_image=query_image,
    top_k=10
)

for r in result.fused_results[:5]:
    print(f"{r.patent_number}: {r.relevance_score:.2%}")
```

## 融合策略详解

### 1. 线性融合 (Linear)
```
Score = w_text × normalize(S_text) + w_image × normalize(S_image)
```

### 2. Reciprocal Rank Fusion (RRF)
```
RRF(d) = Σ 1/(k + rank(d)), k=60
```

### 3. 分数归一化融合
```
Score = w_text × (S - min)/(max - min) + w_image × ...
```
