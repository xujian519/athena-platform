```
发明披露材料 → 发明理解Agent → 结构规划 → 内容生成Agent → 质量审核Agent

```

### 3. 新增类和数据结构

```python
# 撰写阶段
class DraftPhase(Enum):
    INVENTION_UNDERSTANDING = "invention_understanding"
    STRUCTURE_PLANNING = "structure_planning"
    CONTENT_GENERATION = "content_generation"
    QUALITY_CHECK = "quality_check"

# 章节类型
class SectionType(Enum):
    TECHNICAL_FIELD = "technical_field"
    BACKGROUND = "background"
    INVENTION_CONTENT = "invention_content"
    DRAWING_DESCRIPTION = "drawing_description"
    EMBODIMENTS = "embodiments"

# 发明类型
class InventionType(Enum):
    PRODUCT = "product"
    METHOD = "method"
    DEVICE = "device"
    SYSTEM = "system"
    COMPOSITION = "composition"

# 技术特征
@dataclass
class TechnicalFeature:
    feature_id: str
    name: str
    description: str
    is_essential: bool = False
    technical_effect: str = ""
    dependencies: List[str] = field(default_factory=list)

# 发明理解结果
@dataclass
class InventionUnderstanding:
    invention_title: str
    invention_type: InventionType
    technical_field: str
    core_innovation: str
    technical_problem: str
    technical_solution: str
    technical_effects: List[str]
    essential_features: List[TechnicalFeature]
    optional_features: List[TechnicalFeature]

# 章节内容
@dataclass
class SectionContent:
    section_type: SectionType
    title: str
    content: str
    word_count: int = 0
    quality_score: float = 0.0
    issues: List[str] = field(default_factory=list)

# 说明书草稿
@dataclass
class SpecificationDraft:
    draft_id: str
    version: int = 1
    invention_title: str = ""
    sections: Dict[SectionType, SectionContent] = field(default_factory=dict)
    claims: List[str] = field(default_factory=list)
    overall_quality_score: float = 0.0
    iteration_count: int = 0

# 质量报告
@dataclass
class QualityReport:
    overall_score: float
    dimensions: Dict[str, float]
    critical_issues: List[str]
    warnings: List[str]
    suggestions: List[str]
    is_acceptable: bool = False
```

## 使用示例

```python
from core.patent.ai_services.autospec_drafter import (
    AutoSpecDrafter,
    draft_patent_specification,
    format_specification,
    InventionType
)

# 方式1: 便捷函数
draft = await draft_patent_specification(
    disclosure="""
    光伏充电系统

    核心创新：采用智能充电控制算法

    技术效果：
    1. 提高充电效率30%
    2. 延长电池寿命

    主要组件：光伏板、充电控制器、储能电池
    """,
    llm_manager=llm_manager,
    drawing_paths=["/path/to/fig1.png"]
)

# 查看结果
print(f"发明名称: {draft.invention_title}")
print(f"质量评分: {draft.overall_quality_score:.1f}/10")
print(f"迭代次数: {draft.iteration_count}")

# 获取完整说明书
print(format_specification(draft))

# 方式2: 使用撰写框架
drafter = AutoSpecDrafter(llm_manager=llm_manager)
draft = await drafter.draft_specification(
    disclosure="...",
    enable_quality_check=True,
    max_iterations=3
)
```
