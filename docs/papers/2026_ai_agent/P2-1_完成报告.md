### 3. 新增类和数据结构

```python
# 错误类型
class ErrorType(Enum):
    KNOWLEDGE_MISSING = "knowledge_missing"
    KNOWLEDGE_UNUSED = "knowledge_unused"
    KNOWLEDGE_MISAPPLIED = "knowledge_misapplied"
    REASONING_ERROR = "reasoning_error"
    AMBIGUITY = "ambiguity"

# 严重程度
class DiagnosisSeverity(Enum):
    CRITICAL = "critical"
    HIGH = "high"
    MEDIUM = "medium"
    LOW = "low"

# 激活策略
class ActivationStrategy(Enum):
    CLARIFICATION = "clarification"
    SELF_ANSWERING = "self_answering"
    DECOMPOSITION = "decomposition"
    EXAMPLE = "example"
    CHAIN_OF_THOUGHT = "chain_of_thought"

# 澄清问题
@dataclass
class ClarificationQuestion:
    question_id: str
    question: str
    purpose: str
    expected_info: str
    priority: int = 1

# 优化建议
@dataclass
class OptimizationRecommendation:
    recommendation_id: str
    strategy: ActivationStrategy
    description: str
    implementation: str
    expected_improvement: str
    priority: DiagnosisSeverity

# 诊断结果
@dataclass
class DiagnosisResult:
    diagnosis_id: str
    error_type: ErrorType
    severity: DiagnosisSeverity
    error_description: str
    evidence: List[str]
    confidence: float
    clarification_questions: List[ClarificationQuestion]
    recommendations: List[OptimizationRecommendation]

# 激活会话
@dataclass
class ActivationSession:
    session_id: str
    original_query: str
    original_response: str
    diagnosis: Optional[DiagnosisResult]
    improved_response: str
    improvement_score: float
```

## 使用示例

```python
from core.patent.ai_services.knowledge_diagnosis import (
    KnowledgeActivationDiagnoser,
    diagnose_response,
    activate_and_improve,
    format_diagnosis_report,
    ErrorType,
    DiagnosisSeverity
)

# 方式1: 便捷函数
result = await diagnose_response(
    query="发明专利和实用新型专利有什么区别？",
    response="可能是这样，也许有不同...",
    llm_manager=llm_manager
)

# 查看结果
print(f"错误类型: {result.error_type.value}")
print(f"严重程度: {result.severity.value}")

# 格式化报告
print(format_diagnosis_report(result))

# 方式2: 完整激活会话
session = await activate_and_improve(
    query="什么是专利侵权？",
    response="我不太确定...",
    llm_manager=llm_manager
)

print(f"改进回答: {session.improved_response}")
print(f"改进分数: {session.improvement_score:.0%}")
```
