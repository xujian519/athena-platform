## 使用示例

### 7.1 完整功能演示

```python
from core.agent.athena_agent import AthenaAgent

# 创建Agent
agent = AthenaAgent()
await agent.initialize()

# 1. 使用原有功能
result = await agent.process_input("分析专利创造性")
print(f"深度分析: {result['athena_analysis']}")
print(f"技术评估: {result['technical_assessment']}")
print(f"战略建议: {result['strategic_recommendations']}")

# 2. 查看性能数据
print(f"处理时间: {result['performance']['processing_time']:.3f}秒")
print(f"成功率: {result['performance']['statistics']['success_rate']:.1%}")

# 3. 查看路由信息（如果有）
if "routing" in result:
    print(f"意图类型: {result['routing']['intent_type']}")
    print(f"置信度: {result['routing']['confidence']}")

# 4. 记录智慧（如果记忆系统可用）
if agent.memory_enhanced:
    await agent.remember_wisdom(
        content="专利分析需要全面考虑技术方案",
        importance=0.9,
        tags=["专利", "分析"],
    )

    # 回忆知识
    memories = await agent.recall_wisdom("专利分析")
    print(f"相关知识: {len(memories)}条")

# 5. 查看路由统计
if agent.routing_enabled:
    stats = await agent.get_routing_statistics()
    print(f"缓存命中率: {stats['cache_hit_rate']:.1%}")
```
