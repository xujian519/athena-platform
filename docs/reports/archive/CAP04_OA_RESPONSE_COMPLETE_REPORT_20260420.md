## 💻 使用示例

### 基础使用

```python
from core.patent.oa_response import OAResponder, ExaminationOpinion, ResponseOptions

# 创建答复器
responder = OAResponder()

# 定义审查意见
oa = ExaminationOpinion(
    oa_number="OA202312001",
    oa_date="2023-12-01",
    rejection_type="novelty",
    cited_claims=[1, 2],
    prior_art_references=["CN123456789A"],
    examiner_arguments=["对比文件D1公开了相同技术方案"],
    legal_basis=["专利法第22条第2款"]
)

# 当前权利要求
current_claims = [
    "1. 一种图像识别方法，包括输入层和卷积层。",
    "2. 根据权利要求1所述的方法，其特征在于，所述卷积层采用多尺度卷积核。"
]

# 配置选项
options = ResponseOptions(
    include_amendments=True,
    auto_generate_amendments=True,
    writing_style="formal"
)

# 创建答复
response = await responder.create_response(oa, current_claims, options)

# 输出结果
print(f"策略类型: {response.strategy['strategy_type']}")
print(f"成功概率: {response.metadata['success_probability']:.2%}")
print(f"答复理由: {response.response_arguments}")

# 保存到文件
response.save_to_file("答复意见书.txt")
```

### 批量答复

```python
# 批量处理多个审查意见
oa_list = [oa1, oa2, oa3]
claims_dict = {
    "OA202312001": claims1,
    "OA202312002": claims2,
    "OA202312003": claims3
}

responses = await responder.batch_create_responses(
    oa_list,
    claims_dict,
    options=options
)

print(f"成功答复 {len(responses)} 个审查意见")
```
