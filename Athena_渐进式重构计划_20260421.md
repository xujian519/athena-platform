  ```python
  from pathlib import Path
from typing import Dict

def load_env(env_name: str = "development") -> Dict[str, str]:
    """加载环境配置，支持继承"""
    base_env = {}
    current_env = {}

    # 加载基础配置
    if Path(".env").exists():
        with open(".env") as f:
            for line in f:
                line = line.strip()
                if line and not line.startswith("#"):
                    key, value = line.split("=", 1)
                    base_env[key.strip()] = value.strip()

    # 加载环境特定配置
    env_file = Path(f".env.{env_name}")
    if env_file.exists():
        with open(env_file) as f:
            for line in f:
                line = line.strip()
                if line and not line.startswith("#"):
                    key, value = line.split("=", 1)
                    current_env[key.strip()] = value.strip()

    # 合并配置（环境特定配置覆盖基础配置）
    return {**base_env, **current_env}
  ```

  ```python
  from pydantic_settings import BaseSettings, SettingsConfigDict
  from typing import Literal

  class Settings(BaseSettings):
    """统一配置管理"""

    model_config = SettingsConfigDict(
        env_file=".env",
        env_file_encoding="utf-8",
        case_sensitive=False,
    )

    # 环境名
    environment: Literal["development", "test", "production"] = "development"

    # 数据库配置
    database_host: str = "localhost"
    database_port: int = 5432
    database_user: str = "athena"
    database_password: str = "athena123"
    database_name: str = "athena"

    @property
    def database_url(self) -> str:
        return f"postgresql://{self.database_user}:{self.database_password}@{self.database_host}:{self.database_port}/{self.database_name}"

    # Redis配置
    redis_host: str = "localhost"
    redis_port: int = 6379
    redis_password: str = "redis123"
    redis_db: int = 0

    @property
    def redis_url(self) -> str:
        return f"redis://:{self.redis_password}@{self.redis_host}:{self.redis_port}/{self.redis_db}"

    # LLM配置
    llm_provider: Literal["openai", "anthropic", "deepseek"] = "anthropic"
    llm_api_key: str = ""
    llm_model: str = "claude-sonnet-4-6"
    llm_temperature: float = 0.7

    # 加载YAML配置文件
    @classmethod
    def from_yaml(cls, yaml_path: str) -> "Settings":
        """从YAML文件加载配置"""
        with open(yaml_path) as f:
            config_data = yaml.safe_load(f)
        return cls(**config_data)

    # 获取单例
    _instance = None

    @classmethod
    def get_instance(cls) -> "Settings":
        """获取配置单例"""
        if cls._instance is None:
            cls._instance = cls()
        return cls._instance

  # 全局配置实例
  settings = Settings.get_instance()
  ```

- [ ] 实现环境配置加载器
  ```python
  # core/config/env_loader.py
  from typing import Dict, Any
  import yaml
  from pathlib import Path

  def load_config(env: str = "development") -> Dict[str, Any]:
      """加载配置（支持继承）"""
      config = {}

      # 1. 加载基础配置
      base_dir = Path("config/base")
      if base_dir.exists():
          for yaml_file in base_dir.glob("*.yml"):
              with open(yaml_file) as f:
                  config.update(yaml.safe_load(f))

      # 2. 加载环境配置
      env_file = Path(f"config/environments/{env}.yml")
      if env_file.exists():
          with open(env_file) as f:
              config.update(yaml.safe_load(f))

      return config

  def load_service_config(service: str, env: str = "development") -> Dict[str, Any]:
      """加载服务配置"""
      config = load_config(env)

      # 加载服务特定配置
      service_file = Path(f"config/services/{service}.yml")
      if service_file.exists():
          with open(service_file) as f:
              config.update(yaml.safe_load(f))

      return config
  ```

  ```python
  # core/config/validator.py
  from pydantic import BaseModel, validator
  from typing import Dict, Any

  class ConfigValidator(BaseModel):
      """配置验证器"""

    database_url: str
    redis_url: str
    llm_api_key: str

    @validator("database_url")
    def validate_database_url(cls, v):
        if not v.startswith("postgresql://"):
            raise ValueError("Invalid database URL")
        return v

    @validator("redis_url")
    def validate_redis_url(cls, v):
        if not v.startswith("redis://"):
            raise ValueError("Invalid Redis URL")
        return v

    @validator("llm_api_key")
    def validate_llm_api_key(cls, v):
        if not v or len(v) < 10:
            raise ValueError("Invalid LLM API key")
        return v

  @classmethod
  def validate_config(cls, config: Dict[str, Any]) -> bool:
      """验证配置"""
      try:
          cls(**config)
          return True
      except Exception as e:
          print(f"配置验证失败: {e}")
          return False
  ```

  ```python
  from pydantic import BaseModel, Field
  from typing import Dict, List, Optional
  from datetime import datetime
  from enum import Enum

  class ServiceStatus(str, Enum):
      """服务状态"""
      STARTING = "starting"

      STOPPING = "stopping"
      STOPPED = "stopped"
      ERROR = "error"

      status: ServiceStatus
      last_check: datetime
      error_message: Optional[str] = None
      metrics: Dict[str, float] = {}

  class ServiceInfo(BaseModel):
      """服务信息"""
      id: str = Field(..., description="服务ID")
      name: str = Field(..., description="服务名称")
      type: str = Field(..., description="服务类型")
      host: str = Field(..., description="服务地址")
      port: int = Field(..., description="服务端口")
      health_check_url: str = Field(..., description="健康检查URL")
      health: ServiceHealth = Field(default_factory=ServiceHealth)
      metadata: Dict[str, str] = Field(default_factory=dict)
      tags: List[str] = Field(default_factory=list)

  class ServiceRegistry(BaseModel):
  ```

  ```python
  # core/service_registry/registry.py
  from typing import Dict, Optional
  from datetime import datetime, timedelta
  from core.service_registry.models import ServiceInfo, ServiceRegistry, ServiceStatus
  from core.service_registry.health_check import check_health

  class ServiceRegistryManager:
      """服务注册管理器"""

      def __init__(self):
          self.registry = ServiceRegistry()

      async def register(self, service: ServiceInfo) -> bool:
          """注册服务"""
          try:
              # 首次健康检查
              service.health = await check_health(service.health_check_url)

              # 注册服务
              self.registry.services[service.id] = service
              self.registry.last_updated = datetime.now()

              return True
          except Exception as e:
              print(f"注册服务失败: {e}")
              return False

      async def deregister(self, service_id: str) -> bool:
          """注销服务"""
          if service_id in self.registry.services:
              del self.registry.services[service_id]
              self.registry.last_updated = datetime.now()
              return True
          return False

      async def heartbeat(self, service_id: str) -> bool:
          """服务心跳"""
          if service_id not in self.registry.services:
              return False

          service = self.registry.services[service_id]

          # 更新健康状态
          service.health = await check_health(service.health_check_url)

          # 如果服务不健康，移除注册
          if service.health.status != ServiceStatus.RUNNING:
              await self.deregister(service_id)

          return service.health.status == ServiceStatus.RUNNING

      async def health_check_all(self) -> Dict[str, bool]:
          """检查所有服务健康状态"""
          results = {}
          for service_id, service in self.registry.services.items():
              results[service_id] = await self.heartbeat(service_id)
          return results

      def get_all_services(self) -> Dict[str, ServiceInfo]:
          """获取所有服务"""
          return self.registry.services

      def get_service(self, service_id: str) -> Optional[ServiceInfo]:
          """获取特定服务"""
          return self.registry.services.get(service_id)
  ```

  ```python
  # core/logging/formatter.py
  import json
  import logging
  from datetime import datetime
  from typing import Dict, Any

  class JSONFormatter(logging.Formatter):
      """JSON格式化器"""

      def format(self, record: logging.LogRecord) -> str:
          log_data: Dict[str, Any] = {
              "timestamp": datetime.utcnow().isoformat(),
              "level": record.levelname,
              "logger": record.name,
              "message": record.getMessage(),
              "module": record.module,
              "function": record.funcName,
              "line": record.lineno,
          }

          # 添加异常信息
          if record.exc_info:
              log_data["exception"] = self.formatException(record.exc_info)

          # 添加额外字段
          if hasattr(record, "extra_fields"):
              log_data.update(record.extra_fields)

          return json.dumps(log_data, ensure_ascii=False)
  ```

- [ ] 实现统一日志接口
  ```python
  # core/logging/logger.py
  import logging
  from core.logging.formatter import JSONFormatter

  def setup_logger(
      name: str,
      level: str = "INFO",
      log_file: str = None,
      json_format: bool = True
  ) -> logging.Logger:
      """设置日志器"""

      logger = logging.getLogger(name)
      logger.setLevel(getattr(logging, level.upper()))

      # 清除现有处理器
      logger.handlers.clear()

      # 控制台处理器
      console_handler = logging.StreamHandler()
      if json_format:
          console_handler.setFormatter(JSONFormatter())
      else:
          console_handler.setFormatter(
              logging.Formatter(
                  "%(asctime)s - %(name)s - %(levelname)s - %(message)s"
              )
          )
      logger.addHandler(console_handler)

      # 文件处理器
      if log_file:
          file_handler = logging.FileHandler(log_file)
          if json_format:
              file_handler.setFormatter(JSONFormatter())
          else:
              file_handler.setFormatter(
                  logging.Formatter(
                      "%(asctime)s - %(name)s - %(levelname)s - %(message)s"
                  )
              )
          logger.addHandler(file_handler)

      return logger

  # 全局日志器
  logger = setup_logger("athena")
  ```
