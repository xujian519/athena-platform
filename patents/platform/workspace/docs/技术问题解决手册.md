### 3. 数据处理问题

#### 问题：大文件处理内存溢出
```python
# 症状
MemoryError: Unable to allocate array
Killed: 9

# 解决方案
# 1. 分块处理
def process_large_file(filepath, chunk_size=1000):
    with open(filepath, 'r') as f:
        chunk = []
        for i, line in enumerate(f):
            chunk.append(line)
            if len(chunk) >= chunk_size:
                yield chunk
                chunk = []
        if chunk:
            yield chunk

# 2. 流式处理
def stream_vectors(vector_file):
    for line in open(vector_file, 'r'):
        if line.strip():
            yield process_line(line)

# 3. 使用生成器
def lazy_load_embeddings():
    """惰性加载词向量"""
    for word, vector in parse_vector_file():
        yield word, vector
```

### 2. 进程管理
```bash
# 1. 启动脚本
#!/bin/bash
cd /project/src/api
nohup python3 -c "
import uvicorn
from simple_patent_api import app
uvicorn.run(app, host='0.0.0.0', port=8000)
" > /var/log/api.log 2>&1 &
echo $! > /var/run/api.pid

# 2. 停止脚本
#!/bin/bash
if [ -f /var/run/api.pid ]; then
    kill $(cat /var/run/api.pid)
    rm /var/run/api.pid
fi

# 3. 监控脚本
#!/bin/bash
if ! curl -s http://localhost:8000/health > /dev/null; then
    echo "API服务异常，正在重启..."
    ./restart_api.sh
fi
```

### 3. 日志管理
```python
# 1. 结构化日志
import logging
import json
from datetime import datetime

class JSONFormatter(logging.Formatter):
    def format(self, record):
        log_data = {
            "timestamp": datetime.utcnow().isoformat(),
            "level": record.levelname,
            "message": record.getMessage(),
            "module": record.module,
            "function": record.funcName
        }
        return json.dumps(log_data)

# 2. 日志轮转
import logging.handlers
handler = logging.handlers.RotatingFileHandler(
    "api.log", maxBytes=100*1024*1024, backupCount=5
)
```
